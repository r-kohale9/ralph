<!-- Option 2: Load directly -->
<script src="https://storage.googleapis.com/test-dynamic-assets/packages/api-helper/index.js"></script>

<script>
// Initialize API Helper
const api = new APIHelper({
  baseUrl: 'http://localhost:4000',
  timeout: 10000
});

// Configure error handling
api.onError((error) => {
  console.error('API Error:', error);
});

// Submit game results
const gameResults = {
  session_id: 'session_123',
  game_id: 'game_456',
  metrics: { score: 95, time: 120 },
  attempts: [],
  completed_at: Date.now()
};

api.submitResults(gameResults)
  .then(response => console.log('Results submitted:', response))
  .catch(error => console.error('Submission failed:', error));

// Get current configuration
const config = api.getConfig();
console.log('Current config:', config);

// Update configuration
api.configure({
  baseUrl: 'http://localhost:4000',
  headers: {
    'Authorization': 'Bearer token123'
  }
});
</script>
