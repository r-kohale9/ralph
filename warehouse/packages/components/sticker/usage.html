🎨 Sticker Component Demo

A vanilla JavaScript component for displaying animated stickers (Lottie or Images/GIFs) as feedback.

ℹ️ How it works:
Click the buttons below to show different stickers. Stickers auto-hide after the specified duration.

Usage Code

<!-- Include Lottie Player (for Lottie stickers) -->
<script src="https://unpkg.com/@lottiefiles/lottie-player@latest/dist/lottie-player.js"></script>

<!-- Include Sticker Component -->
<script src="sticker-component.js"></script>

<script>
// Basic usage - Lottie
StickerComponent.show({
  type: "LOTTIE",
  json: "https://assets.lottiefiles.com/path/to/animation.json",
  alignment: "RIGHT",
  duration: 3
});

// Image/GIF
StickerComponent.show({
  type: "IMAGE_GIF",
  image: "https://example.com/sticker.gif",
  alignment: "LEFT",
  duration: 3
});

// With callbacks
StickerComponent.configure({
  onShow: function(sticker) {
    console.log('Sticker shown:', sticker);
  },
  onHide: function(sticker) {
    console.log('Sticker hidden:', sticker);
  }
});

// Using stickers list
StickerComponent.configure({
  stickersList: [
    { id: 'success', type: 'LOTTIE', json: {...}, alignment: 'RIGHT' },
    { id: 'error', type: 'IMAGE_GIF', image: 'error.gif', alignment: 'LEFT' }
  ]
});

StickerComponent.show({ sticker: 'success' });

// Hide manually
StickerComponent.hide();

// Get state
const state = StickerComponent.getState();
console.log(state.isVisible);
</script>
