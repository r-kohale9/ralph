Popup Component Demo

This is a vanilla JavaScript popup component that can be imported with a script tag.

Basic Examples

Exact React Usage Example

Usage

<!-- Include Lottie Player -->
<script src="https://unpkg.com/@lottiefiles/lottie-player@latest/dist/lottie-player.js"></script>

<!-- Include Popup Component -->
<script src="popup-component.js"></script>

<script>
// Show popup
PopupComponent.show({
    icon: "https://cdn.homeworkapp.ai/...",
    title: "Permission to play audio",
    description: "Please click okay",
    primaryText: "Okay!",
    primaryClick: () => {
        console.log('Primary clicked');
        PopupComponent.hide();
    },
    hasSecondary: false,
    secondaryText: "Cancel",
    secondaryClick: () => {
        console.log('Secondary clicked');
    }
});

// Hide popup
PopupComponent.hide();

// Check if visible
PopupComponent.isVisible();
</script>
