📢 Subtitle Component Demo

A vanilla JavaScript component for displaying feedback/subtitles at the bottom of the screen.

ℹ️ How it works:
Click the buttons below to show different subtitle messages. Messages auto-hide after the duration you specify.

Usage Code

<!-- Include the script -->
<script src="subtitle-component.js"></script>

<script>
// Basic usage
SubtitleComponent.show({
  text: "Great job!",
  duration: 3
});

// With formatting
SubtitleComponent.show({
  text: "**Bold** and *italic* text",
  duration: 5
});

// With callback
SubtitleComponent.show({
  text: "Processing...",
  duration: 3,
  onHide: function() {
    console.log('Subtitle hidden!');
  }
});

// Hide manually
SubtitleComponent.hide();

// Get current state
const state = SubtitleComponent.getState();
console.log(state.isVisible);

// Configure with custom options
SubtitleComponent.configure({
  dumpLogs: function(data) {
    console.log('Log:', data);
  },
  mixpanelTrack: function(event, props) {
    console.log('Mixpanel:', event, props);
  }
});
</script>
