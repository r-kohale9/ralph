Usage Code

<!-- Single import for all components -->
<script src="https://storage.googleapis.com/test-dynamic-assets/packages/components/index.js"></script>

<script>
// Wait for components to load
window.addEventListener('load', function() {

  // All components are available globally

  // Subtitle
  SubtitleComponent.show({
    text: 'Hello World!',
    duration: 3
  });

  // Sticker
  StickerComponent.show({
    type: 'LOTTIE',
    json: 'https://assets.lottiefiles.com/path/to/animation.json',
    alignment: 'RIGHT',
    duration: 3
  });

  // Timer
  TimerComponent.start({
    duration: 60,
    onComplete: function() {
      console.log('Timer complete!');
    }
  });

  // Popup
  PopupComponent.show({
    title: 'Welcome',
    description: 'Hello from popup!',
    primaryButton: { text: 'OK' }
  });

  // Or use via MathAIComponents namespace
  MathAIComponents.SubtitleComponent.show({...});
  MathAIComponents.StickerComponent.show({...});
  MathAIComponents.TimerComponent.start({...});
  MathAIComponents.PopupComponent.show({...});
});
</script>
