## Pattern 1: Drag and Drop

### HTML

```html
<div class="drag-source" id="drag-source">
  <!-- Draggable items -->
  <div class="drag-item" draggable="true" data-value="5">5</div>
  <div class="drag-item" draggable="true" data-value="+">+</div>
  <div class="drag-item" draggable="true" data-value="3">3</div>
</div>

<div class="drop-zone" id="drop-zone">
  <!-- Drop slots -->
  <div class="drop-slot" data-slot="0"></div>
  <div class="drop-slot" data-slot="1"></div>
  <div class="drop-slot" data-slot="2"></div>
</div>
```

### JavaScript

```javascript
function initDragDrop() {
  document.querySelectorAll('.drag-item').forEach(item => {
    item.addEventListener('dragstart', (e) => {
      e.dataTransfer.setData('text/plain', item.dataset.value);
      item.classList.add('dragging');
    });
    item.addEventListener('dragend', () => {
      item.classList.remove('dragging');
    });
  });

  document.querySelectorAll('.drop-slot').forEach(slot => {
    slot.addEventListener('dragover', (e) => {
      e.preventDefault();
      slot.classList.add('drag-over');
    });
    slot.addEventListener('dragleave', () => {
      slot.classList.remove('drag-over');
    });
    slot.addEventListener('drop', (e) => {
      e.preventDefault();
      slot.classList.remove('drag-over');
      const value = e.dataTransfer.getData('text/plain');
      slot.textContent = value;
      slot.dataset.value = value;
      slot.classList.add('filled');
    });
  });
}

function getDragDropAnswer() {
  return Array.from(document.querySelectorAll('.drop-slot'))
    .map(slot => slot.dataset.value || null);
}

function resetDragDrop() {
  document.querySelectorAll('.drop-slot').forEach(slot => {
    slot.textContent = '';
    delete slot.dataset.value;
    slot.classList.remove('filled', 'correct', 'incorrect');
  });
  document.querySelectorAll('.drag-item').forEach(item => {
    item.classList.remove('used');
  });
}
```

## Pattern 2: Clickable Grid

### HTML

```html
<div class="game-grid" id="game-grid" style="--cols: 4;">
  <!-- Cells generated by JavaScript -->
</div>
```

### CSS

```css
.game-grid {
  display: grid;
  grid-template-columns: repeat(var(--cols), 1fr);
  gap: 4px;
  max-width: 360px;   /* MUST have pixel cap — 100% alone causes huge cells on desktop */
  margin: 0 auto;     /* Center within parent */
}

.grid-cell {
  aspect-ratio: 1;
  display: flex;
  align-items: center;
  justify-content: center;
  border: 2px solid #ddd;
  border-radius: 8px;
  cursor: pointer;
  font-size: var(--mathai-font-size-body);
  font-weight: 600;
  background: #fff;
  transition: all 0.2s ease;
  user-select: none;
  overflow: hidden;    /* Prevent pseudo-elements from escaping cell boundaries */
}

.grid-cell:hover { background: #f0f0f0; }
.grid-cell.selected { border-color: var(--mathai-blue); background: #EBF0FF; }
.grid-cell.correct { border-color: var(--mathai-green); background: var(--mathai-light-green); }
.grid-cell.incorrect { border-color: var(--mathai-red); background: var(--mathai-light-red); }
.grid-cell.disabled { pointer-events: none; opacity: 0.5; }
.grid-cell.path { background: #E8F4FD; border-color: var(--mathai-blue); }
```

### JavaScript

```javascript
function createGrid(rows, cols, data) {
  const grid = document.getElementById('game-grid');
  grid.style.setProperty('--cols', cols);
  grid.innerHTML = '';

  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      const cell = document.createElement('div');
      cell.className = 'grid-cell';
      cell.dataset.row = r;
      cell.dataset.col = c;
      cell.textContent = data[r][c];
      cell.addEventListener('click', () => handleCellClick(r, c, cell));
      grid.appendChild(cell);
    }
  }
}

function handleCellClick(row, col, cell) {
  if (cell.classList.contains('disabled')) return;

  // Toggle selection (single select)
  document.querySelectorAll('.grid-cell.selected').forEach(c => c.classList.remove('selected'));
  cell.classList.add('selected');

  // Or toggle selection (multi select)
  // cell.classList.toggle('selected');
}

function getGridSelections() {
  return Array.from(document.querySelectorAll('.grid-cell.selected'))
    .map(cell => ({ row: +cell.dataset.row, col: +cell.dataset.col, value: cell.textContent }));
}
```

### JavaScript

```javascript
let tags = [];

function handleTagKeydown(event) {
  if (event.key === 'Enter') {
    event.preventDefault();
    const input = document.getElementById('tag-input');
    const value = input.value.trim();
    if (value && !tags.includes(value)) {
      tags.push(value);
      renderTags();
      input.value = '';
    }
  }
}

function removeTag(index) {
  tags.splice(index, 1);
  renderTags();
}

function renderTags() {
  const display = document.getElementById('tags-display');
  display.innerHTML = tags.map((tag, i) =>
    `<span class="tag-chip" data-index="${i}">
      ${tag}
      <span class="tag-remove" onclick="removeTag(${i})">x</span>
    </span>`
  ).join('');
}

function getTagValues() {
  return [...tags];
}

function resetTags() {
  tags = [];
  renderTags();
  document.getElementById('tag-input').value = '';
}
```
